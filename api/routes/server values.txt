create table artworks 
(identifier VARCHAR(255), 
title VARCHAR(255), 
artist VARCHAR(255),
year VARCHAR(4),
theoreticalprice SMALLINT,
actualprice SMALLINT,
hidden BOOL,
owner VARCHAR(255), 
url VARCHAR(255));

create table history 
(identifier VARCHAR(255), 
seller VARCHAR(255), 
buyer VARCHAR(255),
price SMALLINT,
timestamp DATETIME);

create table users 
(username VARCHAR(255), 
hash VARCHAR(255),
name VARCHAR(255), 
admin BOOL,
guilders SMALLINT,
microresearchpoints SMALLINT, 
numofpaintings SMALLINT);

create table microresearch 
(username VARCHAR(255), 
identifier VARCHAR(255), 
information TEXT,
timestamp DATETIME);
