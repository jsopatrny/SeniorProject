create table artworks 
(identifier VARCHAR(255), 
title VARCHAR(255), 
artist VARCHAR(255),
year VARCHAR(4),
theoreticalprice SMALLINT,
actualprice SMALLINT,
owner VARCHAR(255), 
url VARCHAR(255),
rateable BOOL);

create table history 
(identifier VARCHAR(255), 
seller VARCHAR(255), 
buyer VARCHAR(255),
price BIGINT,
timestamp DATETIME,
tradeid BIGINT);

create table users 
(username VARCHAR(255), 
hash VARCHAR(255),
name VARCHAR(255), 
admin BOOL,
guilders SMALLINT,
microresearchpoints SMALLINT, 
blurb TEXT,
formcompleted BOOL);

create table microresearch 
(username VARCHAR(255), 
identifier VARCHAR(255), 
information TEXT,
timestamp DATETIME);

create table trades 
(tradeid BIGINT,  
buyer VARCHAR(255),
seller VARCHAR(255),
buyerinit BOOL,  
sellerinit BOOL,  
buyerapproved BOOL,  
sellerapproved BOOL);

create table tradedetails
(tradeid BIGINT, 
buyer VARCHAR(255),
seller VARCHAR(255),
offer VARCHAR(255),    
approved BOOL,
archived BOOL);

create table auction
(identifier VARCHAR(255),
number BIGINT,
highestbid BIGINT,
username VARCHAR(255),
deadline DATETIME, 
groupid VARCHAR(255),
lotessay TEXT, 
pricevisible BOOL);

// groups is a keyword for mysql >= 8.0.2 - consider renaming
create table groups
(groupid VARCHAR(255),
identifier VARCHAR(255),
date DATETIME, 
allowstudents BOOL);

create table ratetable
(identifier VARCHAR(255), 
price BIGINT);
